def check_sub(user_id):
    try:
        status = bot.get_chat_member(f"@{CHANNEL_USERNAME}", user_id).status
        return status in ['member', 'administrator', 'creator']
    except:
        return True

def sub_markup():

def save_user(user_id):
    with open(USERS_FILE, 'r') as f:
        users = f.read().splitlines()
    if str(user_id) not in users:
        with open(USERS_FILE, 'a') as f:
            f.write(f"{user_id}\n")
        return True
    return False

@bot.message_handler(func=lambda message: 'instagram.com' in message.text)
def handle_insta(message):
    if not check_sub(message.from_user.id):
        return bot.send_message(message.chat.id, "<b>يجب عليك الاشتراك في القناة أولاً.</b>", 
                                parse_mode="HTML", reply_markup=sub_markup())

    status_msg = bot.reply_to(message, "<i>جاري معالجة الطلب.. 📨</i>", parse_mode="HTML")
    url = message.text.split('?')[0]
    
    try:
        shortcode = url.split("/")[-2]
        post = instaloader.Post.from_shortcode(L.context, shortcode)
        
        
        caption_text = post.caption if post.caption else ""
        media_group = []
        
        if post.typename == 'GraphSidecar':
            for i, nodes in enumerate(post.get_sidecar_nodes()):
               
                current_caption = caption_text if i == 0 else ""
                if nodes.is_video:
                    media_group.append(types.InputMediaVideo(nodes.video_url, caption=current_caption))
                else:
                    media_group.append(types.InputMediaPhoto(nodes.display_url, caption=current_caption))
        else:
            if post.is_video:
                media_group.append(types.InputMediaVideo(post.video_url, caption=caption_text))
            else:
                media_group.append(types.InputMediaPhoto(post.url, caption=caption_text))

        if media_group:
            bot.send_media_group(message.chat.id, media_group)
            bot.delete_message(message.chat.id, status_msg.message_id)
            
    except Exception as e:
        print(f"Error detail: {e}")
        bot.edit_message_text("<b>عذراً، فشل التحميل. قد يكون الحساب خاصاً أو هناك مشكلة في الاتصال.</b>", 
                              message.chat.id, status_msg.message_id, parse_mode="HTML")

bot.polling()
